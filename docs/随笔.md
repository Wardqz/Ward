int q[1000010];

void sort1(int a[], int l, int r){
    if(l >= r)  return;
    int mid = (l + r) >> 1, c = l + 1;
    int t = a[mid];
    int tt = a[l]; a[l] = a[mid], a[mid] = tt;
    for(int i = l + 1; i <= r; i ++){
        if(a[i] <= t){
            q[c ++] = a[i];
        }
    }
    int d = c - 1;
    for(int i = l + 1; i <= r; i ++){
        if(a[i] > t) q[c ++] = a[i];
    }
    for(int i = l + 1; i <= r; i ++){
        a[i] = q[i];
    }
    tt = a[l], a[l] = a[d], a[d] = tt;
    
    sort1(a, l, d - 1);
    sort1(a, d + 1, r);
}

void sort(int a[], int len){
    sort1(a, 0, len - 1);
}
